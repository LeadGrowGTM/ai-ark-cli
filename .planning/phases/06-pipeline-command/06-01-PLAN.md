Build the `ai-ark people pipeline` command that chains search → export → find-emails into one invocation. Reuses existing filter flags, poller, profile filter, and persistence layers — no new infrastructure. Users get a fully enriched, email-verified, Tier 1 filtered, persisted contact list from a single command.

Purpose: Eliminate the manual trackId juggling between three async stages — the #1 friction point in v1.0 workflows.

       ```typescript
       const account = buildAccountFilter(filterOpts, "people");

       if (account) searchBody.account = account;
       if (contact) searchBody.contact = contact;

       if (opts.reviewUrl !== false) printReviewUrl(filterOpts, "people");

       if (opts.dryRun) {
         process.stderr.write("Dry run — no pipeline submitted. Search payload:\n");
         formatOutput({ reviewUrl: buildSearchUrl(filterOpts, "people"), request: searchBody }, format);
         return;
       }
       ```

    5. **Stage 1 — Search** (use POST `/people` and capture trackId):
       ```typescript
       process.stderr.write("[1/3] Searching...\n");
       const searchRes = await client.post<PeopleSearchResponse>("/people", searchBody);
       const trackId = searchRes.trackId;
       const searchCount = searchRes.content?.length ?? 0;
       process.stderr.write(`[1/3] Searching... found ${searchCount} profiles (trackId: ${trackId})\n`);
       if (!trackId) {
         console.error("Pipeline failed at stage [1/3]: no trackId returned from search");
         process.exit(1);
       }
       ```

    6. **Stage 2 — Export** (POST `/people/export` with same filters; poll; paginate results):
       ```typescript
       process.stderr.write("[2/3] Exporting...\n");
       const exportBody: ExportPeopleRequest = { page: 0, size: parseInt(opts.size, 10) };
       if (account) exportBody.account = account;
       if (contact) exportBody.contact = contact;
       const exportJob = await client.post<ExportJobResponse>("/people/export", exportBody);
       const exportTrackId = exportJob.trackId;
       const exportPoll = await pollUntilDone(client, `/people/export/${exportTrackId}/inquiries/statistics` as ApiEndpoint);
       if (exportPoll.state === "FAILED") {
         console.error(`Pipeline failed at stage [2/3]: export failed after ${exportPoll.elapsed}s`);
         process.exit(1);
       }
       process.stderr.write(`[2/3] Exporting... done — ${exportPoll.found} records\n`);
       ```

    7. **Stage 3 — Find emails** (use exportTrackId from stage 2):
       ```typescript
       process.stderr.write("[3/3] Finding emails...\n");
       const finderBody: EmailFinderRequest = { trackId: exportTrackId };
       await client.post<EmailFinderStatistics>("/people/email-finder", finderBody);
       const finderPoll = await pollUntilDone(client, `/people/email-finder/${exportTrackId}/statistics` as ApiEndpoint);
       if (finderPoll.state === "FAILED") {
         console.error(`Pipeline failed at stage [3/3]: email finder failed after ${finderPoll.elapsed}s`);
         process.exit(1);
       }
       const allResults: unknown[] = [];
       let page = 0;
       const pageSize = 100;
       while (true) {
         const results = await client.get<EmailFinderResultsResponse>(
           `/people/email-finder/${exportTrackId}/inquiries?page=${page}&size=${pageSize}` as ApiEndpoint,
         );
         allResults.push(...results.content);
         if (results.last || results.content.length === 0) break;
         page++;
       }
       process.stderr.write(`[3/3] Finding emails... done — ${allResults.length} verified\n`);
       ```

       ```typescript
       const filtered = filterByProfile(allResults, "person", profile);

         command: "people-pipeline",
         output: opts.output,
         noSave: opts.save === false,
       });
       if (opts.clayTable) pushToClay(opts.clayTable, filtered as unknown[]);
       formatOutput(filtered, format);
       ```
